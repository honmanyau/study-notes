;; (define (dist-between invader lom) 0)

;; Using template for Invader:
(define (dist-between invader missile)
  (sqrt (+ (sqr (- (invader-x invader) (missile-x missile)))
           (sqr (- (invader-y invader) (missile-y missile))))))
